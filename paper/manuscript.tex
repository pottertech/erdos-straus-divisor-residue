\begin{abstract}
We present a divisor-residue criterion for the Erd\H{o}s--Straus conjecture in the residue class $n \equiv 1 \pmod{12}$, $n \not\equiv 0 \pmod{5}$. We introduce a parameter $A = 4x-n$ and reduce the conjecture to a \textbf{bounded divisor-residue condition}: for each admissible $n$, there must exist bounded $A \equiv 3 \pmod 4$ and a divisor $P$ of $\bigl(n(n+A)/4\bigr)^2$ satisfying
\[
P \equiv -n^2\cdot 4^{-1} \pmod A
\]
with positive $y,z$. We prove that for $n$ prime, $n \equiv 1 \pmod 4$, this condition is equivalent to the target residue $T$ lying in the \textbf{bounded divisor-residue set} $D_A((nx)^2)$, a subset of $(\Z/A\Z)^*$ generated by prime factors of $nx$ with exponents bounded by $2v_p(nx)$. For $A$ prime, this reduces to a quadratic non-residue condition via quadratic reciprocity, conditional on a Bounded Divisor-Residue Lemma that is partially proven and computationally verified. We prove unconditionally that $n \equiv 5 \pmod 8$ always admits $A=3$. We also describe a conditional analytic route, via Burgess-type bounds for least prime quadratic non-residues in arithmetic progressions, that would give
\[
A = O\!\left(n^{1/(4\sqrt e)+\varepsilon}\right).
\]
Computationally, $A\le 31$ covers all 6,666 cases up to $n=100{,}000$, and $A\le 99$ covers all but one of 666,666 cases up to $n=10{,}000{,}000$.
\end{abstract}

\section{Introduction}

The Erd\H{o}s--Straus conjecture states that for every integer $n\ge 2$, there exist positive integers $x,y,z$ such that
\[
\frac{4}{n}=\frac{1}{x}+\frac{1}{y}+\frac{1}{z}.
\]
Despite extensive work since 1950, the conjecture remains open. Mordell (1967) proved that polynomial identities providing solutions for $n\equiv r\pmod m$ exist only when $r$ is a quadratic non-residue modulo $m$. Since $1$ is a quadratic residue modulo every $m$, no polynomial identity can cover $n\equiv 1\pmod m$. This makes $n\equiv 1\pmod 4$---and in particular $n\equiv 1\pmod{12}$---the resistant case.

Mballa (2026) introduced a parametric framework proving density-1 coverage for $n\equiv 1\pmod 4$ using symmetric solutions, requiring a divisor $b\equiv 3\pmod 4$ of $n$. This handles all composite $n$ having a prime factor congruent to $3\pmod 4$, but leaves prime $n\equiv 1\pmod 4$ as the key open case.

Our approach does not use polynomial identities. Instead, we parameterize solutions by $A=4x-n$ and reduce the problem to a \textbf{bounded divisor-residue condition}: a specific residue must be achievable as a product of prime powers of $nx$, with each exponent bounded by the $p$-adic valuation of $nx$. This is a concrete combinatorial condition in elementary number theory, not an algebraic-geometric problem.

\section{The $A$-Parameter Framework}

\subsection{Setup}

For $n\equiv 1\pmod 4$, write
\[
x=\frac{n+A}{4}, \qquad A\equiv 3\pmod 4, \qquad A\ge 3.
\]
The Erd\H{o}s--Straus equation becomes
\[
\frac{4}{n}=\frac{4}{n+A}+\frac{1}{y}+\frac{1}{z}.
\]
Set
\[
m=\frac{n+A}{4}, \qquad nx=n\cdot m.
\]
The remainder $4A/(n(n+A))$ must be decomposed as $1/y+1/z$. The existence of positive $y,z$ is equivalent to finding a divisor $P$ of $(nx)^2$ such that
\begin{enumerate}[label=(\roman*)]
\item $P\equiv -n^2\cdot 4^{-1}\pmod A$, the target residue $T$; and
\item the resulting values
\[
y=\frac{P+nx}{A}, \qquad z=\frac{Q+nx}{A}, \qquad Q=\frac{(nx)^2}{P},
\]
are positive integers.
\end{enumerate}
We call $P$ a \textbf{working divisor} of $(nx)^2$ for the pair $(n,A)$.

\subsection{The Bounded Divisor-Residue Set}

The set of achievable residues $P\pmod A$, where $P$ ranges over divisors of $(nx)^2$, is
\[
D_A((nx)^2)=\left\{\prod_i p_i^{\alpha_i}\bmod A:0\le \alpha_i\le 2v_{p_i}(nx)\right\},
\]
where the product ranges over all primes $p_i$ dividing $nx$, and $v_{p_i}(nx)$ is the $p_i$-adic valuation of $nx$. This is a \textbf{bounded} set: each exponent $\alpha_i$ is limited by $2v_{p_i}(nx)$, reflecting the constraint that $P$ must be an actual divisor of $(nx)^2$.

Note that $D_A((nx)^2)$ is a subset of the subgroup
\[
H(A)=\langle p\bmod A:p\mid nx\rangle \subseteq (\Z/A\Z)^*,
\]
but is generally not equal to $H(A)$, since subgroup membership allows arbitrary integer powers while $D_A$ restricts exponents to bounded ranges. The distinction matters: a subgroup may contain $-1$ even when no actual divisor of $(nx)^2$ realizes the needed residue.

\subsection{The Exact Characterization}

\begin{theorem}[Divisor-Residue Criterion]
For $n$ prime, $n\equiv 1\pmod 4$, and $A\equiv 3\pmod 4$ with $\gcd(n,A)=1$, the parameter $A$ yields an Erd\H{o}s--Straus solution if and only if
\[
T=-n^2\cdot 4^{-1}\pmod A
\]
lies in $D_A((nx)^2)$.
\end{theorem}

\begin{proof}
By construction, $A$ yields a solution if and only if there exists a divisor $P\mid (nx)^2$ such that
\[
P\equiv T\pmod A.
\]
Since $m=(n+A)/4\equiv n\cdot 4^{-1}\pmod A$, we have
\[
T=-n^2\cdot 4^{-1}\equiv -nm\equiv -nx\pmod A.
\]
Thus $P\equiv -nx\pmod A$, so $A\mid P+nx$ and hence
\[
y=\frac{P+nx}{A}
\]
is an integer.

Let $Q=(nx)^2/P$. Because $P\mid (nx)^2$, $Q$ is a positive integer. Also, since $\gcd(nx,A)=1$, the residue $P$ is invertible modulo $A$. From $PQ=(nx)^2$ and $P\equiv -nx\pmod A$, we get
\[
Q\equiv (nx)^2P^{-1}\equiv (nx)^2(-nx)^{-1}\equiv -nx\pmod A.
\]
Therefore $A\mid Q+nx$, and
\[
z=\frac{Q+nx}{A}
\]
is an integer. Since $P,Q,nx>0$, both $y$ and $z$ are positive. Therefore $P\equiv T\pmod A$ with $P\mid (nx)^2$ is exactly the condition for an Erd\H{o}s--Straus solution arising from this $A$.
\end{proof}

\begin{remark}[Verification]
The target may also be written $T=-n\cdot m\pmod A$, since $m\equiv n\cdot 4^{-1}\pmod A$. Computational verification found that $T\in D_A((nx)^2)$ coincides exactly with the existence of Erd\H{o}s--Straus solutions across all 6,666 test cases with $n\le 100{,}000$, $n\equiv 1\pmod{12}$, $n\not\equiv 0\pmod 5$, with zero mismatches.
\end{remark}

\begin{remark}[$D_A$ versus $H(A)$]
In most tested cases, $D_A((nx)^2)=H(A)$. However, $D_A\ne H(A)$ in general; for example, $n=7561$, $A=19$ gives $|D_A|=6$ but $|H(A)|=18$. Despite this, the critical equivalence $-1\in D_A\Longleftrightarrow -1\in H(A)$ holds with zero mismatches across all 6,473 prime-$A$ cases tested. This equivalence is partially proven and computationally verified; see the Bounded Divisor-Residue Lemma below.
\end{remark}

\section{Main Results}

\subsection{The $n\equiv 5\pmod 8$ Case}

\begin{theorem}
For $n$ prime, $n\equiv 5\pmod 8$, and $n\ne 3$, the choice $A=3$ yields an Erd\H{o}s--Straus solution.
\end{theorem}

\begin{proof}
When $n\equiv 5\pmod 8$, we have $n+3\equiv 0\pmod 8$, so
\[
m_3=\frac{n+3}{4}
\]
is even. Therefore $2\mid m_3$, and $2$ is a prime factor of $nx=n\cdot m_3$. Since $2\equiv -1\pmod 3$, take $P=2$.

Because $n$ is prime and $n\ne 3$, we have $n^2\equiv 1\pmod 3$, and since $4^{-1}\equiv 1\pmod 3$,
\[
-n^2\cdot 4^{-1}\equiv -1\equiv 2\pmod 3.
\]
Thus $P=2$ satisfies the target congruence for $A=3$. Since $P\mid (nx)^2$, Theorem 1 gives positive integral $y,z$. Therefore $A=3$ works directly, without invoking the conditional prime-$A$ criterion.
\end{proof}

\subsection{Reduction to a Quadratic Non-Residue Condition}

\begin{theorem}[Prime $A$ Criterion, conditional on the Bounded Divisor-Residue Lemma]
For $A$ prime, $\gcd(n,A)=1$, and $n$ prime with $n\equiv 1\pmod 4$,
\[
T\in D_A((nx)^2)
\quad\Longleftrightarrow\quad
\text{some prime factor }p\mid nx\text{ is a quadratic non-residue modulo }A.
\]
The converse direction is unconditional; the forward direction is conditional and computationally verified in the tested range.
\end{theorem}

\begin{proof}
For $A$ prime, $(\Z/A\Z)^*$ is cyclic of order $A-1$. Since $A\equiv 3\pmod 4$, $(A-1)/2$ is odd, and $-1$ is the unique element of order $2$.

For the converse, suppose no prime factor of $nx$ is a quadratic non-residue modulo $A$. Then all generators of $H(A)$ have odd order, so $h=|H(A)|$ is odd and $-1\notin H(A)$. Since $D_A\subseteq H(A)$, we have $-1\notin D_A$. If $T\in D_A$, then using $T=-nm$ and $nm\in H(A)$, we would have
\[
T\cdot (nm)^{-1}=-1\in H(A),
\]
a contradiction. Hence $T\notin D_A$.

For the forward direction, if some prime $p\mid nx$ is a quadratic non-residue modulo $A$, then $p$ has even order, so $h=|H(A)|$ is even and $-1\in H(A)$. The Bounded Divisor-Residue Lemma below gives $-1\in D_A$ in the proven sub-cases and computationally in the remaining tested cases.

However, $D_A$ is not closed under multiplication. The step from $-1\in D_A$ and $nm\in D_A$ to $T=(-1)(nm)\in D_A$ does not follow automatically, because multiplying two divisor residues may push exponents beyond the bound $2v_p(nx)$. If the divisor realizing $-1$ uses exponent $a_i$ on prime $p_i$, then the product with $nm$ requires
\[
a_i+v_{p_i}(nx)\le 2v_{p_i}(nx),
\]
equivalently $a_i\le v_{p_i}(nx)$.

Define the shifted divisor-residue set
\[
D_A^{(nm)}((nx)^2)=\left\{\prod_i p_i^{\alpha_i}\bmod A:0\le \alpha_i\le 2v_{p_i}(nx)-v_{p_i}(nm)\right\}.
\]
This is the set of residues $R$ such that $R\cdot nm$ is an actual divisor of $(nx)^2$. The correct sufficient condition for the $-1$ factorization route is therefore $-1\in D_A^{(nm)}$, not merely $-1\in D_A$.

Computationally, $T\in D_A((nx)^2)$ has zero mismatches across all 6,473 prime-$A$ cases tested. In 50 of these cases, $-1\notin D_A^{(nm)}$, yet $T\in D_A$ via a different divisor path. Thus a complete proof of the forward direction requires a direct argument for $T\in D_A$, or a full proof that $-1\in D_A^{(nm)}$ when appropriate.
\end{proof}

\subsection{The Bounded Divisor-Residue Lemma}

\begin{lemma}[Bounded Divisor-Residue, partially proven and computationally verified]
For $A$ prime and $\gcd(n,A)=1$, if $-1\in H(A)$, then $-1\in D_A((nx)^2)$ in the proven sub-cases below and in all tested cases.
\end{lemma}

\begin{proof}[Proof sketch and tested classification]
Let $h=|H(A)|$. Since $-1\in H(A)$, $h$ is even. Let $g$ be a primitive root modulo $A$, so $H(A)=\langle g^{(A-1)/h}\rangle$ and $-1=g^{h/2}$ inside $H(A)$.

We must show that $h/2$ lies in the bounded sumset
\[
S=\left\{\sum_i s_i\alpha_i\pmod h:0\le \alpha_i\le 2e_i\right\},
\]
where $s_i$ is the discrete logarithm of $p_i$ in $H(A)$ and $e_i=v_{p_i}(nx)$.

If $h=2$, any quadratic non-residue generator gives $p^1\equiv -1\pmod A$ with exponent $1\le 2e$; this case is proven generally. If some prime factor $p\mid nx$ has $\ord_A(p)=2$, then $p\equiv -1\pmod A$ and again $p^1$ realizes $-1$ directly; this case is also proven generally.

For $h=2m$ with $m>1$ odd and no order-2 quadratic non-residue, decompose $\Z/h\Z\cong \Z/2\Z\times \Z/m\Z$. The $\Z/2\Z$ component is supplied by a single quadratic non-residue. The remaining condition is a bounded sumset condition in $\Z/m\Z$.

Let
\[
S'=A_1+\cdots+A_k\subseteq \Z/m\Z,
\]
where
\[
A_i=\{0,s_i,2s_i,\ldots,2e_i s_i\}.
\]
If $\Stab(S')$ is trivial and
\[
\sum_i |A_i|=\sum_i(2e_i+1)\ge m+k-1,
\]
then Kneser's theorem gives $|S'|\ge m$, hence $S'=\Z/m\Z$ and the required residue is present. This sub-case is proven under the stated hypotheses.

The remaining tested cases are verified computationally. A general proof of the full lemma remains open.
\end{proof}

\begin{corollary}[Conditional sufficient condition]
If $\leg{A}{n}=-1$, then $A$ works, conditional on the forward direction of Theorem 3 and the Bounded Divisor-Residue Lemma. This condition is computationally verified in all tested cases.
\end{corollary}

\begin{proof}
By quadratic reciprocity, since $n\equiv 1\pmod 4$,
\[
\leg{A}{n}=\leg{n}{A}.
\]
If $\leg{A}{n}=-1$, then $n$ is a quadratic non-residue modulo $A$. Since $n$ is a prime factor of $nx$, the conditional forward direction of Theorem 3 applies.
\end{proof}

\subsection{The Composite $n$ Case}

\begin{theorem}[Mballa 2026]
If $n$ has a prime factor $p\equiv 3\pmod 4$, then $A=p$ yields an Erd\H{o}s--Straus solution.
\end{theorem}

This handles all composite $n\equiv 1\pmod 4$ having a prime factor congruent to $3\pmod 4$. Combined with Theorem 2, the only remaining case is prime $n\equiv 1\pmod 8$.

\subsection{Legendre Symbol Identity for the $m$-Route}

\begin{theorem}[Legendre Symbol Identity]
For any prime $p$ dividing $m=(n+A)/4$, the Legendre symbol satisfies
\[
\leg{p}{A}=\leg{n}{p}.
\]
Thus $p$ is a quadratic non-residue modulo $A$ if and only if $n$ is a quadratic non-residue modulo $p$.
\end{theorem}

\begin{proof}
Since $p\mid m=(n+A)/4$, we have $A\equiv -n\pmod p$. By quadratic reciprocity, since $n\equiv 1\pmod 4$,
\[
\leg{p}{A}=\leg{A}{p}(-1)^{((p-1)/2)((A-1)/2)}.
\]
Since $A\equiv 3\pmod 4$, $(A-1)/2$ is odd. If $p\equiv 1\pmod 4$, then
\[
\leg{p}{A}=\leg{A}{p}=\leg{-n}{p}=\leg{-1}{p}\leg{n}{p}=\leg{n}{p}.
\]
If $p\equiv 3\pmod 4$, then
\[
\leg{p}{A}=-\leg{A}{p}=-\leg{-n}{p}=-\leg{-1}{p}\leg{n}{p}=\leg{n}{p}.
\]
Thus the identity holds in both cases.
\end{proof}

\begin{corollary}[Route formulation]
A works if either $\leg{A}{n}=-1$ (the direct route), conditional as above, or some prime $p\mid (n+A)/4$ has $\leg{n}{p}=-1$ (the $m$-route), again subject to the same bounded divisor-residue condition. Both routes are governed by the same quadratic character of $n$.
\end{corollary}

\subsection{Computational Verification}

\begin{theorem}[Computational, $n\le 100{,}000$]
For all 6,666 integers $n\equiv 1\pmod{12}$, $n\not\equiv 0\pmod 5$, with $13\le n\le 100{,}000$, there exists
\[
A\in\{3,7,11,15,19,23,27,31\}
\]
yielding an Erd\H{o}s--Straus solution. The maximum $A$ required is $31$.
\end{theorem}

\begin{theorem}[Computational, $n\le 10{,}000{,}000$]
For all 666,666 integers $n\equiv 1\pmod{12}$, $n\not\equiv 0\pmod 5$, with $13\le n\le 10{,}000{,}000$, there exists $A\le 99$ yielding an Erd\H{o}s--Straus solution, with one exception: $n=8{,}803{,}369$, which requires $A=107$. The maximum $A$ required excluding this outlier is $59$.
\end{theorem}

The 666,666 count refers to all integers in the residue class, not just primes. Only prime $n$ need be tested after reductions, but the verification included all integers for completeness.

\subsection{Analytic Bound}

\begin{proposition}[Analytic route via Burgess-type bounds]
Conditional on a standard least-prime-QNR-in-arithmetic-progression estimate: for $n$ prime, $n\equiv 1\pmod 4$, there exists a prime $A\equiv 3\pmod 4$ with $\leg{A}{n}=-1$ and
\[
A=O\!\left(n^{1/(4\sqrt e)+\varepsilon}\right)
\]
for any $\varepsilon>0$. Such an $A$ yields an Erd\H{o}s--Straus solution subject to the conditional forward direction of Theorem 3.
\end{proposition}

\begin{proof}[Proof sketch]
The Burgess bound applied to character sums over primes $p\le X$ with $p\equiv 3\pmod 4$ suggests that the character sum
\[
\sum_{\substack{p\le X\\ p\equiv 3\pmod 4}}\leg{p}{n}
\]
is small relative to $\pi(X;4,3)$ for $X=n^{1/(4\sqrt e)+\varepsilon}$ under the needed least-prime-QNR-in-AP estimate. A fully sourced proof requires careful citation of the prime number theorem in arithmetic progressions combined with Burgess-type character sum bounds. We state this as a conditional analytic route, not as an unconditional theorem.
\end{proof}

Under GRH, the least quadratic non-residue modulo $n$ is $O((\log n)^2)$, and the least prime congruent to $3\pmod 4$ that is a quadratic non-residue modulo $n$ is also expected to be $O((\log n)^2)$ under the corresponding prime-in-AP formulation.

\section{The Covering Cascade}

The covering set
\[
\{3,7,11,15,19,23,31\}
\]
verified for $n\le 100{,}000$ operates via two mechanisms.

\textbf{Direct route.} For most prime cases up to $100{,}000$, at least one $A\in\{7,11,19,23,31\}$ has $\leg{A}{n}=-1$, and that $A$ works directly in the tested range. The case $A=3$ always has $\leg{3}{n}=1$ when $n\equiv 1\pmod{12}$.

\textbf{$m$-route.} For the 21 tested cases where $n$ is a quadratic residue modulo all of $7,11,19,23,31$, the $m$-route provides a solution: the smallest quadratic non-residue $p$ of $n$ appears as a prime factor of $m_A$ for some $A$ in the covering set, and $\leg{p}{A}=\leg{n}{p}=-1$ by Theorem 5.

\section{The Constant Bound Conjecture}

\begin{conjecture}
There exists an absolute constant $C$ such that for every prime $n\equiv 1\pmod 4$, there exists $A\le C$ with $A\equiv 3\pmod 4$ and
\[
T\in D_A((nx)^2).
\]
\end{conjecture}

Evidence includes:
\begin{itemize}
\item $A\le 31$ for all $n\le 100{,}000$ in the tested residue class;
\item $A\le 99$ for all but one of 666,666 cases up to $n=10{,}000{,}000$;
\item $A\le 107$ for all tested cases up to $n=10{,}000{,}000$;
\item a Burgess-type least-prime-QNR-in-AP estimate would give $A=O(n^{0.152\ldots})$, conditionally;
\item GRH gives a conditional $O((\log n)^2)$ route.
\end{itemize}

This conjecture is stronger than what GRH implies. Proving it would require showing that among a finite set of primes, at least one is always a quadratic non-residue modulo $n$, or proving an equivalent bounded divisor-residue statement.

\section{Context from Known Results}

\subsection{Mordell (1967)}
Mordell's theorem rules out polynomial identities for $n\equiv 1\pmod m$. The present framework uses divisor-residue conditions, not polynomial identities.

\subsection{Elsholtz--Tao (2013)}
Elsholtz and Tao showed that the total number of solutions satisfies an asymptotic of the form $\sum_{p\le N} f(p)\asymp N\log^2 N$. This provides context on solution density.

\subsection{Vaughan (1970) and Webb (1970)}
Vaughan showed that potential counterexamples have density zero. Webb showed that only a sublinear number of values need more than two terms. These results are consistent with the computational evidence here.

\subsection{Bright--Loughran (2020)}
Bright and Loughran showed that the Brauer--Manin obstruction does not explain failure of the integral Hasse principle for Erd\H{o}s--Straus surfaces.

\subsection{Mballa (2026)}
Mballa introduced a parametric framework proving density-1 coverage for $n\equiv 1\pmod 4$ via symmetric solutions. The present framework gives an exact divisor-residue criterion and a separate unconditional proof for the $n\equiv 5\pmod 8$ case.

\subsection{Burgess (1963)}
Burgess-type character sum bounds are the key input for the conditional analytic route described above. The application to primes in arithmetic progressions with a prescribed quadratic character requires careful sourcing.

\subsection{Salez (2014)}
Salez verified the Erd\H{o}s--Straus conjecture computationally up to $10^{17}$. The present work provides a structural divisor-residue framework for explaining why small $A$ values frequently succeed.

\section{Open Problems}

\begin{enumerate}
\item \textbf{Bounded Divisor-Residue Lemma in full generality.} Prove that $-1\in H(A)$ implies $-1\in D_A((nx)^2)$, or characterize the precise exceptions.

\item \textbf{$D_A$ closure and shifted divisor-residue set.} Since $D_A$ is not closed under multiplication, prove $T\in D_A$ directly whenever a quadratic non-residue prime factor exists, or prove the appropriate shifted-set statement.

\item \textbf{Constant bound conjecture.} Prove $A\le C$ for some absolute constant.

\item \textbf{Covering system proof.} Prove that $\{3,7,11,15,19,23,31\}$ covers all $n$, not just $n\le 100{,}000$.

\item \textbf{Full proof of Proposition 8.} Provide a fully sourced proof combining Burgess bounds with prime number theorem estimates in arithmetic progressions for the least prime $\equiv 3\pmod 4$ that is a quadratic non-residue modulo $n$.

\item \textbf{$m$-route sufficiency.} Prove that when the direct route fails, the $m$-route always succeeds for some $A$ in the covering set.
\end{enumerate}
